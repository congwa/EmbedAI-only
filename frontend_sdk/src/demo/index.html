EmbedAI 聊天窗SDK

智能商品推荐聊天机器人，一行代码轻松集成

🚀 快速开始

将以下代码添加到您的网页中：

<script src="https://cdn.embedai.com/sdk/latest/intellirecs-ai.js" defer></script>
<script>
  window.IntelliRecsAI.init({
    apiBase: "http://localhost:8000",
    theme: { 
      primary: "#2563eb", 
      corner: "16px" 
    },
    welcome: "你好，我是你的智能导购助手。想找点什么？",
    lang: "zh-CN",
    productLinkMode: "new_tab",
    enableHistory: true
  });
</script>

🔧 配置选项

const config = {
  // 必需配置
  apiBase: "https://api.embedai.com", // API基础地址
  
  // 主题配置
  theme: {
    primary: "#2563eb",              // 主色调
    corner: "16px",                  // 圆角大小
    fontFamily: "system-ui",         // 字体
    fontSize: "14px"                 // 字体大小
  },
  
  // 功能配置
  welcome: "欢迎使用智能助手",        // 欢迎语
  lang: "zh-CN",                    // 语言
  productLinkMode: "new_tab",       // 商品链接打开方式
  enableHistory: true,              // 启用历史记录
  maxHistoryLength: 20,             // 最大历史长度
  
  // 事件回调
  callbacks: {
    onOpen: () => console.log('聊天窗打开'),
    onClose: () => console.log('聊天窗关闭'),
    onSend: (message) => console.log('发送消息:', message),
    onReceive: (response) => console.log('收到回复:', response),
    onProductClick: (product) => console.log('点击商品:', product),
    onError: (error) => console.error('发生错误:', error)
  }
};
